ES5核心技巧

1. 变量提升

    <script>
        function test(){
            console.log(a);//结果是undefined,因为变量提升
            var a = 1;
        }
        test();
        console.log(a); //结果是a is not defined 因为变量只提升到函数的顶部



        function test2(){
            console.log(a);
            var a = 1;
            function a(){

            }
        }
        test2();//结果是方法体a,因为函数提升的优先级要比变量高。当发现变量名和方法名一样，但是变量还没有赋值的时候，就直接把变量给忽略了。


        function test3(){
            console.log(a);
            if(false){
                var a = 1;
            }
        }
        test3();//结果是undefined,因为变量提升是提升到方法内的顶端，并不是提升到代码快的顶端。
    </script>


    <script>
        (function(){
            var a = b = 1;
            //相当于: var a = 1;b = 1;  b是全局变量，a 出不了方法区。
        })()
        alert(a);//a is not defined
        alert(b);//b = 1;
    </script>


    JS内部有一个VO对象和AO对象，VO用来存储提升的数据，AO用来激活存储的数据。

3. this指向

        var a = 20;
        var show  = {
            a:10,
            method:function(){
                console.log(this.a);
            }
        }
        show.method();//这里是show对象调用的method方法，所以this指的是show对象，this.a == show.a;
        var fun = show.method;
        fun();//这里的结果是20，因为method方法赋值给了fun，而fun最终是window调用的，所以this指的是window

        var show2 = {
            a:10,
            method:function(){
                function test(){
                    console.log(this.a);
                }
                test();
            }
        }
        show2.method();//这里的结果是20，this指的是window，因为里面的test方法不知道是调用的，
        //所以默认为window。

        var show3 = {
            a:10,
            method: ()=> {
                console.log(this.a);
            }
        }
        show3.method();//这里的结果是20，因为在ES6的箭头函数中，this就是继承父类的this，相当于bind()

4. 按值传递和按引用传递

    <script>
        function test(m){
            m = {
                n:1
            }
        }
        var m = {
            k:10
        }
        test(m);
        console.log(m.n);//结果是undefined，因为m是对象，对象在堆中开辟空间，创建新的地址值，在
        //test方法中，相当于重新创建了一个m，而我们所打印的m是第一次就创建好的m。
    </script>
